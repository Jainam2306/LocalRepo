The FizzBuzz Challenge

A classic programming exercise — counting to 100, replacing multiples of 3 with "Fizz", multiples of 5 with "Buzz", and both with "FizzBuzz".

Source Code
FizzBuzz.js 
var output = [];
var count = 1;

function fizzBuzz() {

  while (count <= 100) {

    if (count % 3 === 0 && count % 5 === 0) {
      output.push("FizzBuzz");

    } else if (count % 3 === 0) {
      output.push("Fizz");

    } else if (count % 5 === 0) {
      output.push("Buzz");

    } else {
      output.push(count);
    }

    count = count + 1;
  }

  console.log(output);
}

fizzBuzz();
